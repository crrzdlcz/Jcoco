// Producci贸n principal.
Coco ::= ( funcion )* <INICIO> ( sentencia )* <FINAL> <EOF> 
// Funciones, van antes de la funcion principal (< INICIO > *** < FINAL >)
// Si regresa un valor, debe ser especificado explicitamente,
// declarando el tipo y el nombre de la variable
funcion ::= <INICIO_DECLARACION_FUNCION> <IDENTIFICADOR> <FLECHA> tiposDeDatos <PARENTESIS_ABRE> listaParametros <PARENTESIS_CIERRA> bloqueDeCodigo 
// Lista de parametros que se pueden pasar a una funci贸n. (Producci贸n)
listaParametros ::= ( parametro ( <COMA> parametro )* )? 
// Regla que define los parametros individuales.
parametro ::= tiposDeDatos <DOS_PUNTOS> <IDENTIFICADOR> 
// Es lo que esta dentro de llaves
// - > Su "scope" debe ser 1.
bloqueDeCodigo ::= <LLAVE_ABRE> ( sentencia )* <LLAVE_CIERRA> 
// Producci贸n sentencia
sentencia ::= declaracionDeVariable
| declaracionVector
| salida
| entrada
| bucleFor
| bucleWhile
| condicionalIf
| condicionalSwitch
| sentenciaRetorno
| sentenciasConIdentificador
sentenciasConIdentificador ::= llamadaFuncion <PUNTO_Y_COMA>
| accesoVector <OPERADOR_ASIGNACION> expresion <PUNTO_Y_COMA>
| <IDENTIFICADOR> <OPERADOR_ASIGNACION> expresion <PUNTO_Y_COMA>
sentenciaRetorno ::= <RETURN> expresion <PUNTO_Y_COMA>
llamadaFuncion ::= <IDENTIFICADOR> <PARENTESIS_ABRE> ( expresion ( <COMA> expresion )* )? <PARENTESIS_CIERRA> 
// *** Expresiones usando precedencia ***

// Expresiones (+, -, ||)
expresion ::= expresionRelacional ( ( <OPERADOR_SUMA> | <OPERADOR_RESTA> | <OPERADOR_OR> ) expresionRelacional )* 
// relacionales (==, !=, <, >, <=, >=)
expresionRelacional ::= termino ( ( <OPERADOR_IGUALDAD> | <OPERADOR_DESIGUALDAD> | <OPERADOR_MENOR_QUE> | <OPERADOR_MAYOR_QUE> | <OPERADOR_MENOR_O_IGUAL> | <OPERADOR_MAYOR_O_IGUAL> ) termino )? 
// Terminos (*, /, %, &&)
termino ::= factor ( ( <OPERADOR_MULTIPLICACION> | <OPERADOR_DIVISION> | <OPERADOR_MODULO> | <OPERADOR_AND> ) factor )* 
// Factores (!, ++, --)
factor ::= <NUMERO_ENTERO>
| <NUMERO_FLOTANTE>
| <CADENA_DE_CARACTERES>
| <IDENTIFICADOR>
| <PARENTESIS_ABRE> expresion <PARENTESIS_CIERRA>
| <OPERADOR_NOT> factor
| <OPERADOR_INCREMENTO> factor
| <OPERADOR_DECREMENTO> factor
| <OPERADOR_RESTA> factor 
// Fin de las expresiones
declaracionDeVariable ::= <INICIO_DECLARACION_VARIABLE> tiposDeDatos <DOS_PUNTOS> <IDENTIFICADOR> ( <OPERADOR_ASIGNACION> ( expresion ) )? <PUNTO_Y_COMA> 
// print!("Hola mundo");
salida ::= <SALIDA> <PARENTESIS_ABRE> ( expresion ) <PARENTESIS_CIERRA> <PUNTO_Y_COMA> 
// scan!(operando);
entrada ::= <ENTRADA> <PARENTESIS_ABRE> <IDENTIFICADOR> <PARENTESIS_CIERRA> <PUNTO_Y_COMA> 
// Vector
declaracionVector ::= tiposDeDatos <IDENTIFICADOR> dimensionVector <PUNTO_Y_COMA>
dimensionVector ::= <CORCHETE_ABRE> ( <NUMERO_ENTERO> | <IDENTIFICADOR> ) <CORCHETE_CIERRA>
accesoVector ::= <IDENTIFICADOR> <CORCHETE_ABRE> ( <NUMERO_ENTERO> | <IDENTIFICADOR> ) <CORCHETE_CIERRA> 
// FIN vector


// Bucles.
// - > For
bucleFor ::= <BUCLE_FOR> <PARENTESIS_ABRE> ( inicializacionFor )? <PUNTO_Y_COMA> ( expresion )? <PUNTO_Y_COMA> ( actualizacionFor )? <PARENTESIS_CIERRA> bloqueDeCodigo 
// Declaracion e inicializacion de variables en bucles.
inicializacionFor ::= declaracionDeVariable
| asignacionSinPuntoYComa 
// Auxiliar
asignacionSinPuntoYComa ::= <IDENTIFICADOR> <OPERADOR_ASIGNACION> expresion
| accesoVector <OPERADOR_ASIGNACION> expresion
actualizacionFor ::= asignacionSinPuntoYComa ( <COMA> asignacionSinPuntoYComa )* 
// - > While
bucleWhile ::= <BUCLE_WHILE> <PARENTESIS_ABRE> expresion <PARENTESIS_CIERRA> bloqueDeCodigo 
// FIN bucles.


// Condicionales.
// - > If
condicionalIf ::= <CONDICIONAL_IF> <PARENTESIS_ABRE> expresion <PARENTESIS_CIERRA> bloqueDeCodigo ( <CONDICIONAL_ELSE> ( bloqueDeCodigo | condicionalIf ) )? 
// - > Swich
condicionalSwitch ::= <CONDICIONAL_SWITCH> <PARENTESIS_ABRE> expresion <PARENTESIS_CIERRA> <LLAVE_ABRE> ( caseSwitch )* ( defaultSwitch )? <LLAVE_CIERRA>
caseSwitch ::= <CASE_CONDICIONAL_SWITCH> ( <NUMERO_ENTERO> | <CADENA_DE_CARACTERES> ) <DOS_PUNTOS> ( sentencia )* ( sentenciaBreak )?
sentenciaBreak ::= <BREAK> <PUNTO_Y_COMA>
defaultSwitch ::= <DEFAULT_SWITCH> <DOS_PUNTOS> ( sentencia )* ( sentenciaBreak )? 
// FIN condicionales.
tiposDeDatos ::= <ENTERO>
| <FLOTANTE>
| <BOOLEANO>
| <CHAR>
| <STRING>
